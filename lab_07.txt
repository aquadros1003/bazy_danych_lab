1.SELECT w.nazwa,COUNT(u.id_uczestnika), GROUP_CONCAT(k.nazwa SEPARATOR ' ')
FROM wyprawa AS w, uczestnicy AS u, kreatura AS k
WHERE w.id_wyprawy=u.id_wyprawy AND u.id_uczestnika=k.idKreatury
GROUP BY w.nazwa;
